// ✅ GAS 後端（保持你原本邏輯 + CORS） const SPREADSHEET_ID = "1YtievhvypvP_0SktKoBjtANU_StEPlk_L1bgcpnl5UE"; function doPost(e) { const params = JSON.parse(e.postData.contents); const email = params.email; const unit = params.targetUnit; const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName("工作表1"); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == unit && data[i][1] == email) { return output({status:"success", url:data[i][2]}); } } return output({status:"error", message:"權限不足"}); } function output(obj){ return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
